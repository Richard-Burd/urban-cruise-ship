import SolutionDropdown from '/components/SolutionDropdown.js'

export const problem = "Carbon Leakage";
export const solution = "Border Adjustment";

Carbon pricing can put domestic manufacturing at a disadvantage and be undermined by carbon leaking. We propose a border adjustment to remedy this problem. A border adjustment would place a tariff on imported goods that is equal to the difference between the domestic carbon price and the price at which the goods are charged in their home country.

  <SolutionDropdown 
    problem={problem} 
    solution={solution}
  >

  </SolutionDropdown>
